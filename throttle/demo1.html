      function throttle(fn, delay) {
        let last = 0
        return function () {
          const now = Date.now()
          if (now - last > delay) {
            fn.apply(this, arguments)
            last = now
          }
        }
      }

      let count = 0
      function addCount() {
        counterDom.textContent = ++count
      }

      const throttled = throttle(addCount, 2000)
