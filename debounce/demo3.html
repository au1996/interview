      function debounce(fn, delay, immediate) {
        let timer = null
        let debounced = function () {
          if (timer) {
            clearTimeout(timer)
          }
          if (immediate) {
            let callNow = !timer
            timer = setTimeout(() => {
              timer = null
            }, delay)
            if (callNow) {
              fn.apply(this, arguments)
            }
          } else {
            timer = setTimeout(() => {
              fn.apply(this, arguments)
            }, delay)
          }
        }
        debounced.cancel = function () {
          clearTimeout(timer)
          timer = null
        }

        return debounced
      }
