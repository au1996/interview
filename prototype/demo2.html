每个引用类型都有一个__proto__属性，并且指向它的构造函数的prototype原型对象 
      const obj = {}
      const arr = []
      const fn = function () {}

      console.log(obj.__proto__ === Object.prototype) // true
      console.log(arr.__proto__ === Array.prototype) // true
      console.log(fn.__proto__ === Function.prototype) // true
