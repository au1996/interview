      function Foo() {
        var getName = function () {
          console.log(1)
        }
        return this
      }

      function getName() {
        console.log(5)
      }

      Foo.prototype.getName = function () {
        console.log(3)
      }

      Foo().getName() // 5
      new Foo().getName() //3
